class Solution {
public:
    bool checkValidString(string s) {
        vector<int> preo(s.size()) , pres(s.size()) , sufc(s.size()) , sufs(s.size());
        preo[0] = (s[0] == '(') - (s[0] == ')');
        pres[0] = (s[0] == '*');
        sufc[s.size()-1] = (s[s.size()-1] == ')') - (s[s.size()-1] == '(');
        sufs[s.size()-1] = (s[s.size()-1] == '*');
        for(int i = 1 ; i < s.size() ; i++) {
            preo[i] = preo[i-1] + (s[i] == '(') - (s[i] == ')');
            pres[i] = pres[i-1] + (s[i] == '*');
        }
        for(int i = s.size()-2 ; i >= 0 ; i--) {
            sufc[i] = sufc[i+1] + (s[i] == ')') - (s[i] == '(');
            sufs[i] = sufs[i+1] + (s[i] == '*');
        }
        for(int i = 0 ; i < s.size() ; i++) if(preo[i] + pres[i] < 0) return false;
        for(int i = s.size()-1 ; i >= 0 ; i--) if(sufc[i] + sufs[i] < 0) return false;
        return true;
    };
};
